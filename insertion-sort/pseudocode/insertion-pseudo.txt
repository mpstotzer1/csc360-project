void insertion_sort(input array, input array_size){

	for (array_index = 1; array_index < array_size; array_index++){
		current_value = array[array_index];
		left_of_index = array_index - 1;
	
		while (left_of_index >= 0 && array[left_of_index] > current_value){
			array[left_of_index + 1] = array[left_of_index];
			left_of_index = left_of_index - 1;
		}

		array[left_of_index + 1] = current_value;
	}
}
